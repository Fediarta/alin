import numpy as np
A = np.array([[6,2],[4,2]])
B = np.array([[3,7],[4,8]])
C = np.array([[1,0],[0,1]])
D = np.array([[1,0,0],[0,1,0],[0,0,1]])
F = np.array([[1,4,2],[1,1,1],[3,2,1]])



det_A = np.linalg.det(A)
det_B = np.linalg.det(B)
det_C = np.linalg.det(C)
det_D = np.linalg.det(D)
det_F = np.linalg.det(F)




print("Matriks 2x2")

print("")
print(f"matrik C:\n{C}")
print(f"Determinan Matrix C :\n{det_C}")
print("")
print("Matriks 3x3  ")
print(f"matrik D:\n{D}")
print(f"Determinan Matrix D :\n{det_D}")
print("")
print(f"matrik F:\n{F}")

import numpy as np

i = np.array([[4, 4, 6], [2, 6, 2], [4, 8, 4]])

DETERMINAN MATRIKS DENGAN KOFAKTOR#

\(Q=\begin{bmatrix} 4 & 3 & 1 & 6\\ 2 & 3 & 2 & 1\\ 3 & 2 & 1 & 1\\ 2 & 5 & 1 & 4 \end{bmatrix} Rumus-determinan= A11 K11 + A12 K12 + A13 K13 + A14 K14 \)

dari matriks buat kofaktor terlebih dahulu#

\( Kij= (-1)^i+j | Mij | \)

  • arti dari rumus diatas adalah baris ke i dan kolom ke j dari matriks akan dihilangkan

\(Q=+4\begin{bmatrix} 3 & 2 & 1 \\ 2 & 1 & 1 \\ 5 & 1 & 4 \end{bmatrix} \) \(-3\begin{bmatrix} 2 & 2 & 1 \\ 3 & 1 & 1 \\ 2 & 1 & 4 \end{bmatrix} \) \(+1\begin{bmatrix} 2 & 3 & 1 \\ 3 & 2 & 1 \\ 2 & 5 & 4 \end{bmatrix} \) \(-6\begin{bmatrix} 2 & 3 & 2 \\ 3 & 2 & 1 \\ 2 & 5 & 1 \end{bmatrix} \)

mencari kofaktor 11#

\(\begin{bmatrix} 3 & 2 & 1 \\ 2 & 1 & 1 \\ 5 & 1 & 4 \end{bmatrix} \) \(+3\begin{bmatrix} 1 & 1 \\ 1 & 4 \end{bmatrix} \) \(-2\begin{bmatrix} 2 & 1 \\ 5 & 4 \end{bmatrix} \) \(+1\begin{bmatrix} 2 & 1 \\ 5 & 1 \end{bmatrix} \)

\( Rumus-determinan= A11 K11 + A12 K12 + A13 K13 + A14 K14 \) \( maka = +3 (1 4 - 1 1)+ (-2) (2 4 - 1 5)+ 1 (2 1 - 1 5) \) \( = +3(3) + (-2)(3) + 1(-3) \) \( = 9 + (-6) + (-3) \) \( =0 \)

mencari kofaktor 12#

\(\begin{bmatrix} 2 & 2 & 1 \\ 3 & 1 & 1 \\ 2 & 1 & 4 \end{bmatrix} \) \(+2\begin{bmatrix} 1 & 1 \\ 1 & 4 \end{bmatrix} \) \(-2\begin{bmatrix} 3 & 1 \\ 2 & 4 \end{bmatrix} \) \(+1\begin{bmatrix} 3 & 1 \\ 2 & 1 \end{bmatrix} \)

\( Rumus-determinan= A11 K11 + A12 K12 + A13 K13 + A14 K14 \) \( maka = +2 (1 4 - 1 1)+ (-2) (3 4 - 1 2)+ 1 (3 1 - 1 2) \) \( = +2(3) + (-2)(10) + 1(1) \) \( = 6 + (-20) + (1) \) \( = -13 \)

mencari kofaktor 13#

\(\begin{bmatrix} 2 & 3 & 1 \\ 3 & 2 & 1 \\ 2 & 5 & 4 \end{bmatrix} \) \(+2\begin{bmatrix} 2 & 1 \\ 5 & 4 \end{bmatrix} \) \(-3\begin{bmatrix} 3 & 1 \\ 2 & 4 \end{bmatrix} \) \(+1\begin{bmatrix} 3 & 2 \\ 2 & 5 \end{bmatrix} \)

\( Rumus-determinan= A11 K11 + A12 K12 + A13 K13 + A14 K14 \) \( maka = +2 (2 4 - 1 5)+ (-3) (3 4 - 1 2)+ 1 (3 5 - 2 2) \) \( = +2(3) + (-3)(10) + 1(11) \) \( = 6 + (-30) + 11 \) \( = -13 \)

mencari kofaktor 14#

\(\begin{bmatrix} 2 & 3 & 2 \\ 3 & 2 & 1 \\ 2 & 5 & 1 \end{bmatrix} \) \(+2\begin{bmatrix} 2 & 1 \\ 5 & 1 \end{bmatrix} \) \(-3\begin{bmatrix} 3 & 1 \\ 2 & 1 \end{bmatrix} \) \(+2\begin{bmatrix} 3 & 2 \\ 2 & 5 \end{bmatrix} \)

\( Rumus-determinan= A11 K11 + A12 K12 + A13 K13 + A14 K14 \) \( maka = +2 (2 1 - 1 5)+ (-3) (3 1 - 1 2)+ 2 (3 5 - 2 2) \) \( = +2(-3) + (-3)(1) + 2(11) \) \( = (-6) + (-3) + 22 \) \( = 13 \)

hasil akhir#

\(Q=\begin{bmatrix} 4 & 3 & 1 & 6\\ 2 & 3 & 2 & 1\\ 3 & 2 & 1 & 1\\ 2 & 5 & 1 & 4 \end{bmatrix} Rumus-determinan= A11 K11 + A12 K12 + A13 K13 + A14 K14 \)

\( det Q = 4 (0) + (-3) (-13) + 1 (-13) + -(6) (-13) \) \( = 0 + 39 + (-13) + (-78) \) \( = 39 + (-91)\) \( = - 52 \)
